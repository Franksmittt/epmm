        function loadImage(input, targetId) {
            if (input.files && input.files[0]) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    const img = new Image();
                    img.onload = function() {
                        const canvas = document.createElement('canvas');
                        const MAX_SIZE = 1080; 
                        let width = img.width;
                        let height = img.height;

                        if (width > height) {
                            if (width > MAX_SIZE) {
                                height *= MAX_SIZE / width;
                                width = MAX_SIZE;
                            }
                        } else {
                            if (height > MAX_SIZE) {
                                width *= MAX_SIZE / height;
                                height = MAX_SIZE;
                            }
                        }
                        canvas.width = width;
                        canvas.height = height;
                        const ctx = canvas.getContext('2d');
                        ctx.drawImage(img, 0, 0, width, height);

                        const safeDataUrl = canvas.toDataURL('image/jpeg', 0.9);
                        const container = document.getElementById(targetId);
                        container.style.backgroundImage = `url(${safeDataUrl})`;
                        container.style.backgroundSize = 'cover';
                    };
                    img.src = e.target.result;
                }
                reader.readAsDataURL(input.files[0]);
            }
        }

        function downloadAd(targetId) {
            const container = document.getElementById(targetId);
            const button = event.target;
            const originalText = button.innerText;
            
            button.innerText = "Rendering...";
            button.disabled = true;

            htmlToImage.toPng(container, { 
                pixelRatio: 2.16, // Outputs exactly 1080x1080
                width: 500,       // Strict Canvas Width
                height: 500,      // Strict Canvas Height
                backgroundColor: '#111111',
                style: {
                    transform: 'scale(1)', 
                    transformOrigin: 'top left',
                    margin: '0'
                }
            })
            .then(function (dataUrl) {
                const link = document.createElement('a');
                link.download = `ATC_${targetId}_1080x1080_Studio.png`;
                link.href = dataUrl;
                link.click();
                
                button.innerText = originalText;
                button.disabled = false;
            })
            .catch(function (error) {
                console.error("Engine Error: ", error);
                alert("Render failed. Make sure you have uploaded an image first.");
                button.innerText = originalText;
                button.disabled = false;
            });
        }
